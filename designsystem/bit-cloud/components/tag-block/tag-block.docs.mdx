import { TagBlock } from './tag-block';

## Basic usage

Use the tag block component:

```js
<TagBlock />
```

### Display the text

```js live
<TagBlock
    heading="Lorem ipsum"
    //icon="SalesLaw"
    invertedColors={false}
    secondaryArrowStyle={false}
    tags={[
    { titleInNavigation: "Lorem" },
    { titleInNavigation: "Ipsum" },
    { titleInNavigation: "Dolor" },
    { titleInNavigation: "Sit" }
    ]}
/>
```
