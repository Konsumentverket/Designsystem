import { FormRadiobutton } from './form-radiobutton';

## Basic usage

Use the form radiobutton component:

```js
<FormRadiobutton />
```

```js live
<FormRadiobutton
    name="lorem-radio-docs"
    id="lorem-radio-docs"
    labelText="Lorem ipsum"
    value="Lorem ipsum"
    onChange={(e) => console.log(e.target.value)}
/>
```
