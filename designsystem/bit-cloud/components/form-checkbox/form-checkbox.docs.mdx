import { FormCheckbox } from './form-checkbox';

## Basic usage

Use the form checkbox component:

```js
<FormCheckbox />
```

### Display the text

```js live
<FormCheckbox
    id="lorem-check-docs"
    name="lorem-check-docs"
    value="Lorem ipsum"
    labelText="Lorem ipsum"
    onChange={(e) => console.log(e.target.value)}
/>
```
