import { Heading } from './heading';

## Basic usage

Use the heading component:

```js
<Heading />
```

```js live
<Heading level={1} text="h1 Lorem ipsum" headingLine />
```

```js live
<Heading level={3} text="h3 Lorem ipsum" />
```

```js live
<Heading level={1} text="h1 with styleLevel 4" styleLevel={4} />
```
